// Typing text

const text="Will you be my Valentine? 💕";

(function type(){

  }

})();

// NO button floats here & there 😈

const no=document.getElementById("no");

function moveNo(){

  no.style.left=Math.random()*(innerWidth-120)+"px";

  no.style.top=Math.random()*(innerHeight-60)+"px";

}

no.addEventListener("mouseover",moveNo);

no.addEventListener("touchstart",moveNo);

// Page switch

function go(n){

  document.querySelectorAll(".page").forEach(p=>p.classList.remove("active"));

  document.getElementById("p"+n).classList.add("active");

}

// YES magic

function yesClick(){

  explodeName();

  showRing();

  setTimeout(()=>go(2),1300);

}

// Floating hearts

setInterval(()=>{

  const h=document.createElement("div");

  h.className="heart";

  h.innerHTML="❤️";

  h.style.left=Math.random()*100+"vw";

  h.style.fontSize=18+Math.random()*22+"px";

  document.body.appendChild(h);

  setTimeout(()=>h.remove(),6000);

},500);

// Name explosion

function explodeName(){

  for(let k=0;k<30;k++){

    const s=document.createElement("div");

    s.className="spark";

    s.innerHTML="Ritu 💖";

    s.style.left="50%";

    s.style.top="50%";

    s.style.setProperty("--x",(Math.random()*300-150)+"px");

    s.style.setProperty("--y",(Math.random()*300-150)+"px");

    document.body.appendChild(s);

    setTimeout(()=>s.remove(),1000);

  }

}

// Ring animation

function showRing(){

  const r=document.createElement("div");

  r.className="ring";

  r.innerHTML="💍";

  r.style.position="absolute";

  r.style.left="50%";

  r.style.top="50%";

  r.style.transform="translate(-50%,-50%)";

  document.body.appendChild(r);

  setTimeout(()=>r.remove(),1200);

}
